N1:=3;
N2:=21;

短期:=100*(C-LLV(L,N1))/(HHV(C,N1)-LLV(L,N1));
长期:=100*(C-LLV(L,N2))/(HHV(C,N2)-LLV(L,N2));

深V1:=REF(短期,2)>90 AND REF(短期,1)<=20 AND 短期>90;
深V2:=REF(短期,3)>90 AND REF(短期,2)<=20 AND 短期>90;
深V出现:=(深V1 OR 深V2);

浅V1:=REF(长期,2)>90 AND REF(长期,1)>=80 AND 长期>90;
浅V2:=REF(长期,3)>90 AND REF(长期,2)>=80 AND 长期>90;
浅V出现:=(浅V1 OR 浅V2);

选股条件: (深V出现 AND 浅V出现);
